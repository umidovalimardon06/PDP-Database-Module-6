*Sequence - bu database obyekti bo'lib,ma'lum bir unique yoki non
            unique integer(butun-son)larni generate qilib beruvchi tip.

	CREATE SEQUENCE [NAME] 
	[DEFAULT EXPRESSION]
	START WITH (N)
	INCREMENT BY (M)
	MINVALUE (L)
	MAXVALUE	 (1)
	CYCLE | NO CYCLE

	CREATE TABLE [TABLE-NAME](
	ID INT PRIMARY KEY DEFALUT nexval('sequence-name')
	)
