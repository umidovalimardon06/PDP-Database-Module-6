Data-types in PostgreSql:

1.Integers:
	smallint(int2)
	integer(int4)
	bigint(int8)

2.Arbitrary-precision:
	decimal/numeric

	id numeric(5,2) ->  6.99, 431.18;
	id numeric -> unlimited;

3.Floating-point:
	real(float4) 6 decimal;
	double precision(float8) 15 decimal;

4.Serial-types:
	smallserial(2byte)
	serial 
	bigserial

5.Monetary-types:
	money(8byte) decimal 2 fixed

6.Character-types:
	character varying(n) / varchar(n)  - flexible
	character(n) / char(n)  - fixed lenght

7.Binary-data-type:
	bytea

8.Date/Time-type:
	timestamp - date and time without time zone
	timestampz - with time zone
	date - calendar data
	interval - time span

	YEAR
	MONTH
	DAY
	HOUR
	MINUTE
	SECOND
	YEAR TO MONTH
	DAY TO HOUR
	DAY TO MINUTE
	DAY TO SECOND
	HOUR TO MINUTE
	HOUR TO SECOND
	MINUTE TO SECOND

9.Boolean-type:
	boolean/bool - true,false,null
		       1,0,t,f

10.Enumerated-type:
	enum.

11.Text-search-types:
	tsvector
	tsquery

12.UUID-type:
	uuid.

13.XML-type:
	xml.

14.JSON-type:
	json
	jsonb

15.Arrays:
	any type can be defined as array;
